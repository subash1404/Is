import java.util.Scanner; class CeaserCipherEnc { static String encrypt(String msg, int key) { String encryptMsg = ""; int encryptChar; msg = msg.toUpperCase(); for (int i = 0; i < msg.length(); i++) { encryptChar = (((int) msg.charAt(i) + key - 65) % 26) + 65; encryptMsg = encryptMsg + (char) encryptChar; } return encryptMsg; } public static void main(String[] args) { String plainText, cipherText; Scanner sc = new Scanner(System.in); System.out.print("Enter a message: "); plainText = sc.nextLine(); System.out.print("Enter a key: "); int userKey = sc.nextInt(); cipherText = encrypt(plainText, userKey); System.out.println("Encrypted message: " + cipherText); sc.close(); } }
